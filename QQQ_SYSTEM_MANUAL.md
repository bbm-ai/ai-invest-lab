const SHEET_ID = 'YOUR_SHEET_ID';

function doPost(e) {
  const action = e.parameter.action;
  const data = JSON.parse(e.parameter.data);
  
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let result = { success: false };
  
  switch(action) {
    case 'daily_log':
      result = writeDailyLog(ss, data);
      break;
    case 'validation':
      result = writeValidation(ss, data);
      break;
    case 'weekly_review':
      result = writeWeeklyReview(ss, data);
      break;
    case 'factor_scores':
      result = writeFactorScores(ss, data);
      break;
    case 'risk_event':
      result = writeRiskEvent(ss, data);
      break;
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const action = e.parameter.action;
  const ss = SpreadsheetApp.openById(SHEET_ID);
  
  if (action === 'history') {
    const days = parseInt(e.parameter.days) || 7;
    const data = getHistory(ss, days);
    return ContentService.createTextOutput(JSON.stringify(data))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService.createTextOutput(JSON.stringify({error: 'Unknown action'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function writeDailyLog(ss, data) {
  const sheet = ss.getSheetByName('DailyLog');
  sheet.appendRow([
    data.date,
    data.market_data?.close,
    data.market_data?.change_pct,
    data.market_data?.vix,
    data.scoring?.total_score,
    data.scoring?.regime,
    data.allocation?.qqq_pct,
    data.allocation?.cash_pct,
    data.prediction?.next_day_bias,
    data.strategy,
    new Date().toISOString()
  ]);
  return { success: true, message: 'Daily log saved' };
}

function writeValidation(ss, data) {
  const sheet = ss.getSheetByName('Validation');
  sheet.appendRow([
    data.date,
    data.prediction_date,
    data.predicted_direction,
    data.actual_direction,
    data.actual_change_pct,
    data.is_correct,
    data.pnl_pct,
    data.pnl_amount,
    data.strategy,
    new Date().toISOString()
  ]);
  return { success: true, message: 'Validation saved' };
}

function writeWeeklyReview(ss, data) {
  const sheet = ss.getSheetByName('WeeklyReview');
  sheet.appendRow([
    data.week_start,
    data.week_end,
    data.trading_days,
    data.week_return,
    data.qqq_return,
    data.alpha,
    data.win_rate,
    data.profit_loss_ratio,
    data.max_drawdown,
    data.prediction_accuracy,
    data.ending_nav,
    new Date().toISOString()
  ]);
  return { success: true, message: 'Weekly review saved' };
}

function writeFactorScores(ss, data) {
  const sheet = ss.getSheetByName('FactorScores');
  const date = data.date;
  const strategy = data.strategy;
  
  for (const [factor, scoreData] of Object.entries(data.factor_scores)) {
    sheet.appendRow([
      date,
      factor,
      scoreData.score,
      data.weights?.[factor] || 0,
      scoreData.direction,
      strategy,
      new Date().toISOString()
    ]);
  }
  return { success: true, message: 'Factor scores saved' };
}

function writeRiskEvent(ss, data) {
  const sheet = ss.getSheetByName('RiskEvents');
  sheet.appendRow([
    data.date,
    data.event_type,
    data.trigger_value,
    data.threshold,
    data.action_taken,
    new Date().toISOString()
  ]);
  return { success: true, message: 'Risk event saved' };
}

function getHistory(ss, days) {
  const sheet = ss.getSheetByName('DailyLog');
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const rows = data.slice(1).slice(-days);
  
  return rows.map(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    return obj;
  });
}
